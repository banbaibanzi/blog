  drop: ['console','debugger'], // 移除所有console和debugger

js

  pure: ['console.log'],  // 移除 console.log

压缩 ​

1.gzip文件压缩 ​

插件 vite-plugin-compression

文档链接：vite-plugin-compression

安装插件

js
npm i vite-plugin-compression -D
1

vite.config.js配置

js
plugins: [
  viteCompression({
    verbose: true, // 是是否在控制台输出压缩结果 默认true
    disable: false, // 是否禁用 默认false
    threshold: 10240, // 体积大于 threshold 才会被压缩,单位 b 10240=10k
    algorithm: 'gzip',  // 压缩算法,可选 [ 'gzip' , 'brotliCompress' ,'deflate' , 'deflateRaw']
    ext: '.gz', // 生成的压缩包后缀
    deleteOriginFile: true, // 压缩后是否删除源文件
  })
]
